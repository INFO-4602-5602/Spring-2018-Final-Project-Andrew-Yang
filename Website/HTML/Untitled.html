dfDP = df[(df.Award == 'Cinematography')]
by_DP = dfDP.groupby('Film')
topDP = pd.DataFrame()
topDP['Wins'] = by_DP['Winner'].sum()
topDP['WinRate'] = by_DP['Winner'].mean()
topDP = topDP.sort_values('Wins', ascending=False) 
topDP['Director of Photography'] = topDP.index
#print(topact)
topact = topDP.head(20)
ax = sns.barplot(x='Wins', y='Director of Photography', color = 'gold',  
            data=topact.head(10))
plt.xlabel('Awards');
plt.ylabel('Cinematographer');
plt.title('Top 10 Oscar Winning Cinematographers');
